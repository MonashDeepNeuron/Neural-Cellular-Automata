import React, { useEffect, useRef, useState } from 'react';

// Import your shaders and utilities
import { guiShader } from './guiShader';
import { computeShader } from './computeShader';
import BufferManager from './managers/BufferManager';
import DeviceManager from './managers/DeviceManager';
import EventManager from './managers/EventManager';
import { parseRuleString, displayRule } from './Parse';
import startingPatterns from './startingPatterns';

const CellularAutomata: React.FC = () => {
    const canvasRef = useRef<HTMLCanvasElement>(null);
    const animationRef = useRef<number | null>(null);

    // State for toggling play/pause
    const [isRunning, setIsRunning] = useState(false);

    
    // WebGPU resources
    const deviceRef = useRef<GPUDevice | null>(null);
    const contextRef = useRef<GPUCanvasContext | null>(null);
    const pipelinesRef = useRef({
        renderPipeline: null as GPURenderPipeline | null,
        computePipeline: null as GPUComputePipeline | null,
    });

    const bindGroupsRef = useRef<any>(null);
    const stepRef = useRef(0);

    // Initialization and Cleanup
    useEffect(() => {
        const initializeWebGPU = async () => {
            if (!canvasRef.current) return;

            // Initialize Device and Context
            await DeviceManager.staticConstructor();
            const device = DeviceManager.device;
            const canvas = DeviceManager.canvas!;
            const context = canvas.getContext('webgpu')!;
            const canvasFormat = navigator.gpu.getPreferredCanvasFormat();

            context.configure({
                device,
                format: canvasFormat,
            });

            deviceRef.current = device;
            contextRef.current = context;

            // Create Buffers, Shaders, and Pipelines
            const { vertexBuffer, vertexBufferLayout } = BufferManager.loadShapeVertexBuffer(device, new Float32Array([
                -0.8, -0.8, -0.8, 0.8, 0.8, 0.8, 0.8, 0.8, 0.8, -0.8, -0.8, -0.8,
            ]));

            const cellShaderModule = device.createShaderModule({ code: guiShader });
            const simulationShaderModule = device.createShaderModule({ code: computeShader });

            const pipelineLayout = device.createPipelineLayout({
                bindGroupLayouts: [BufferManager.createBindGroupLayout(device)],
            });

            const renderPipeline = device.createRenderPipeline({
                layout: pipelineLayout,
                vertex: { module: cellShaderModule, entryPoint: 'vertexMain', buffers: [vertexBufferLayout] },
                fragment: { module: cellShaderModule, entryPoint: 'fragmentMain', targets: [{ format: canvasFormat }] },
            });

            const computePipeline = device.createComputePipeline({
                layout: pipelineLayout,
                compute: { module: simulationShaderModule, entryPoint: 'computeMain' },
            });

            pipelinesRef.current = { renderPipeline, computePipeline };
            bindGroupsRef.current = BufferManager.initializeBuffers(device);

            // Initial Render
            renderPass();
        };

        initializeWebGPU();

        return () => {
            // Cleanup
            if (animationRef.current) cancelAnimationFrame(animationRef.current);
        };
    }, []);

    // Render Pass Function
    const renderPass = () => {
        if (!deviceRef.current || !contextRef.current || !pipelinesRef.current.renderPipeline) return;

        const device = deviceRef.current;
        const context = contextRef.current;
        const encoder = device.createCommandEncoder();
        const pass = encoder.beginRenderPass({
            colorAttachments: [{
                view: context.getCurrentTexture().createView(),
                loadOp: 'clear',
                clearValue: { r: 0, g: 0, b: 0, a: 1 },
                storeOp: 'store',
            }],
        });

        pass.setPipeline(pipelinesRef.current.renderPipeline!);
        pass.draw(6, 1); // Replace with appropriate vertex count
        pass.end();
        device.queue.submit([encoder.finish()]);
    };

    // Update Loop Function
    const updateLoop = () => {
        if (!deviceRef.current) return;

        const device = deviceRef.current;
        const encoder = device.createCommandEncoder();

        if (isRunning) {
            // Perform compute pass and render pass
            const computePass = encoder.beginComputePass();
            computePass.setPipeline(pipelinesRef.current.computePipeline!);
            computePass.dispatchWorkgroups(16, 16); // Example grid size
            computePass.end();
        }

        renderPass();
        device.queue.submit([encoder.finish()]);

        animationRef.current = requestAnimationFrame(updateLoop);
    };

    // Toggle Play/Pause
    const togglePlayPause = () => {
        setIsRunning((prev) => !prev);
        if (!isRunning) {
            animationRef.current = requestAnimationFrame(updateLoop);
        } else if (animationRef.current) {
            cancelAnimationFrame(animationRef.current);
            animationRef.current = null;
        }
    };

    return (
        <div>
            <canvas ref={canvasRef}></canvas>
            <button onClick={togglePlayPause}>{isRunning ? 'Pause' : 'Play'}</button>
        </div>
    );
};

export default CellularAutomata;
